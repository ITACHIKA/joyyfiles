from sklearn.datasets import make_blobs

centers = [[1, 1], [-1, -1], [1, -1]]
x, y = make_blobs(n_samples=1000, centers=centers, cluster_std=0.5, random_state=123)

plt.scatter(x[:, 0], x[:, 1])
plt.show()

from sklearn.cluster import KMeans

kmmodel=KMeans(n_clusters=3,max_iter=500)

y_pred=kmmodel.fit_predict(x)

#print(x)

#print(y)
#print(y_pred)

print(rand_score(y,y_pred))
print(adjusted_rand_score(y,y_pred))

res=np.unique(y_pred)

print(res)

plt.figure()

for clust in res:
    row=np.where(y_pred==clust)
    plt.scatter(x[row,0],x[row,1])
